<script type="text/javascript">
/* 图片预加载兼容函数，适配2018旧版Edge */
function preloadImg(src, callback) {
    var img = new Image();
    img.onload = function(){ callback(); };
    img.onerror = function(){ console.log("图片加载失败："+src); };
    img.src = src;
}

/* 文本容器自适应文字大小，自动缩放字体适配页面留白 */
function autoFit(el, wrap) {
    var wrapH = wrap.clientHeight;
    var wrapW = wrap.clientWidth;
    var fs = 16;
    el.style.fontSize = fs + "px";
    while (el.scrollHeight <= wrapH && el.scrollWidth <= wrapW && fs < 60) {
        fs++;
        el.style.fontSize = fs + "px";
    }
}

/* 荷花背景遮罩加载逻辑，文字渲染完成后再显示背景 */
function loadBg(maskDom) {
    preloadImg("https://djm.pages.dev/b/lotus.png", function(){
        maskDom.style.display = "block";
    });
}

/* 通用远程文本加载函数：加载txt文本、自适应文字、加载背景、异常报错 */
async function loadText(url, textEl, maskEl) {
    try {
        var res = await fetch(url);
        if (!res.ok) throw new Error("文件读取失败");
        var text = await res.text();
        textEl.className = "text-content";
        textEl.textContent = text;
        autoFit(textEl, textEl.closest(".text-container"));
        loadBg(maskEl);
    } catch (err) {
        textEl.className = "text-content error-tip";
        textEl.textContent = "网页解析失败，可能是不支持的网页类型，请检查网页或稍后重试";
        loadBg(maskEl);
    }
}
